def cache_file(url: str, fname: str, dir_: str = './__cache') -> str:
    """Cache the file at given url in the given dir_ with the given
    fname and return the local path.

    Preconditions:
    - dir_ exists
    """
    local_path = f'{dir_}/{fname}'
    if fname not in os.listdir(dir_):
        r = requests.get(url, allow_redirects=True)
        open(local_path, 'wb').write(r.content)
    return local_path

Prepare the data

Initialise a new DataFrame using data, with the column dose as an ordered Categorical index to ensure that the data will be sorted as expected.

cat_data = pd.DataFrame(
    index=pd.Index(
        pd.Categorical(data['dose'], doses, ordered=True),
        name='dose'
    )
).assign(
    n_cases=data['n_cases'].to_numpy(),
    n_controls=data['n_controls'].to_numpy()

Dose-specific odds and odds ratio

Get data with the odds and log(odds) of disease.

def with_odds(df: pd.DataFrame) -> pd.DataFrame:
    return df.assign(
        odds=df['n_cases'].div(df['n_controls']),
        log_odds=np.log(df['n_cases'].div(df['n_controls']))
    )

cat_data.pipe(with_odds)

Get data with the odds ratios.

def with_odds_ratio(df: pd.DataFrame, ref_dose: str) -> pd.DataFrame:
    return df.assign(
        odds_ratio=(df['odds'].to_numpy()
                    / float(df.query('dose == @ref_dose')['odds']))
    )

cat_data.pipe(with_odds).pipe(with_odds_ratio, ref_dose)
